{\bf\dates{2012\,--}
Current research, with Ian Agol}
\begin{itemize}

\item Studying various problems in 3-manifold topology. Currently working on
using hierarchies of Haken 3-manifolds to prove nontriviality of the
Reshetikhin--Turaev invariants on such manifolds.

\end{itemize}

{\bf \dates{2010 Summer}
Langton's Ant, with
Richard Schwartz}
\begin{itemize}
\item Studied the asymptotic and computational properties of the Langton's Ant
dynamical system, 
proved various entropic and measure-theoretic propositions, and 
obtained conjectures of ergodic behavior and periodic
point classification which can be generalized to related systems.
\end{itemize}

{\bf\dates{2009 Summer}
Norms on $\mathbb{R}^2$, with
Justin Peters}
\begin{itemize}
\item Researched the norms on $\mathbb{R}^2$ as a function space with intrinsic
properties, obtained multiple results concerning the extreme points and Choquet
integrals over extreme points in a symmetric subspace of the norm space, and
generalized these results to other symmetry groups and to higher dimensions.
\end{itemize}
\end{jobs}
